<script>
const searchInput = document.getElementById("searchInput");
const typeFilter = document.getElementById("typeFilter");
const table = document.getElementById("transactionTable");

function filterTable(){
    const filter = searchInput.value.toLowerCase();
    const type = typeFilter.value;
    const rows = table.querySelectorAll("tbody tr");

    rows.forEach(row => {
        const text = row.textContent.toLowerCase();
        const rowType = row.children[2].textContent;

        const matchesSearch = text.includes(filter);
        const matchesType = !type || rowType === type;

        row.style.display = (matchesSearch && matchesType) ? "" : "none";
    });
}

searchInput.addEventListener("keyup", filterTable);
typeFilter.addEventListener("change", filterTable);

// Modal
function openModal(){
    document.getElementById("transactionModal").style.display = "flex";
}

function closeModal(){
    document.getElementById("transactionModal").style.display = "none";
}

// Add transaction
function addTransaction(){
    const date = document.getElementById("tDate").value;
    const item = document.getElementById("tItem").value;
    const type = document.getElementById("tType").value;
    const qty = document.getElementById("tQty").value;
    const staff = document.getElementById("tStaff").value;

    if(!date || !item || !qty || !staff){
        alert("Fill all fields");
        return;
    }

    const tbody = table.querySelector("tbody");
    const row = document.createElement("tr");

    row.innerHTML = `
        <td>${date}</td>
        <td>${item}</td>
        <td class="${type === 'IN' ? 'type-in' : 'type-out'}">${type}</td>
        <td>${qty}</td>
        <td>${staff}</td>
    `;

    tbody.appendChild(row);
    closeModal();
}
</script>

<script>
const searchInput = document.getElementById("searchInput");
const typeFilter = document.getElementById("typeFilter");
const table = document.getElementById("transactionTable");

function filterTable(){
    const filter = searchInput.value.toLowerCase();
    const type = typeFilter.value;
    const rows = table.querySelectorAll("tbody tr");

    rows.forEach(row => {
        const text = row.textContent.toLowerCase();
        const rowType = row.children[2].textContent;

        const matchesSearch = text.includes(filter);
        const matchesType = !type || rowType === type;

        row.style.display = (matchesSearch && matchesType) ? "" : "none";
    });
}

searchInput.addEventListener("keyup", filterTable);
typeFilter.addEventListener("change", filterTable);

// Modal
function openModal(){
    document.getElementById("transactionModal").style.display = "flex";
}

function closeModal(){
    document.getElementById("transactionModal").style.display = "none";
}

// Add transaction
function addTransaction(){
    const date = document.getElementById("tDate").value;
    const item = document.getElementById("tItem").value;
    const type = document.getElementById("tType").value;
    const qty = document.getElementById("tQty").value;
    const staff = document.getElementById("tStaff").value;

    if(!date || !item || !qty || !staff){
        alert("Fill all fields");
        return;
    }

    const tbody = table.querySelector("tbody");
    const row = document.createElement("tr");

    row.innerHTML = `
        <td>${date}</td>
        <td>${item}</td>
        <td class="${type === 'IN' ? 'type-in' : 'type-out'}">${type}</td>
        <td>${qty}</td>
        <td>${staff}</td>
    `;

    tbody.appendChild(row);
    closeModal();
}
</script>
